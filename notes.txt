1. composer create-project laravel/laravel ticketing_app
2. php artisan make:model Kategori -m
3. php artisan make:seeder UserSeeder
4. php artisan make:controller Admin/LocationController --resource
4. php artisan db:seed --class=CategorySeeder
5. php artisan db:seed
6. php artisan make:middleware AdminMiddleware
7. ->withMiddleware(function (Middleware $middleware) {
    $middleware->alias([
        'admin' => \App\Http\Middleware\AdminMiddleware::class,
    ]);
})
8. php artisan make:controller Admin/CategoryController --resource
